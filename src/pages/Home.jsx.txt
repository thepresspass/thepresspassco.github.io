import React, { useCallback } from 'react';
import MarqueeTicker from '../components/landing/MarqueeTicker';
import CornerNav from '../components/landing/CornerNav';
import HeroSection from '../components/landing/HeroSection';
import ScoutingReport from '../components/landing/ScoutingReport';
import AuthorSection from '../components/landing/AuthorSection';
import FinalWhistle from '../components/landing/FinalWhistle';
import ProgressScrubber from '../components/landing/ProgressScrubber';

export default function Home() {
  const handleNavigate = useCallback((label) => {
    const sectionMap = {
      "DRAFT DATE": "hero",
      "THE AUTHOR": "author",
      "THE CHAPTERS": "chapters",
      "JOIN THE ROSTER": "join",
    };
    const id = sectionMap[label];
    if (id) {
      document.getElementById(id)?.scrollIntoView({ behavior: 'smooth' });
    }
  }, []);

  return (
    <div className="bg-background min-h-screen">
      <CornerNav onNavigate={handleNavigate} />
      <ProgressScrubber />

      <MarqueeTicker variant="top" />
      <HeroSection />
      <ScoutingReport />
      <AuthorSection />
      <FinalWhistle />
      <MarqueeTicker variant="bottom" />

      {/* Footer micro */}
      <div className="py-6 px-6 md:px-12 flex items-center justify-between border-t border-border/20">
        <span className="font-mono text-[10px] tracking-[0.2em] text-muted-foreground/30">
          © 2026 DANIEL ALSTON
        </span>
        <span className="font-mono text-[10px] tracking-[0.2em] text-muted-foreground/30">
          THE PRESS PASS
        </span>
      </div>
    </div>
  );
}
